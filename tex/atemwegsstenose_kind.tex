    \end{itemize}
  }
  
  \noteWithNote{Pseudokrupp}{\color{cDrug} $\rightarrow$ SDH+Suprarenin}{%
    \begin{itemize}
      \item weniger \textit{krank}, kein/niedriges Fieber, langsamer Beginn,\\
        häufig abends/nachts im Herbst, typisch bellender Husten
    \end{itemize}
  }
  \noteWithNote{Epiglottitis}{\color{cDrug} $\rightarrow$SDH+Suprarenin}{%
    \begin{itemize}
      \item schwer krank, hohes Fieber, plötzlicher Beginn, ganzjährig, \\
        ausgeprägte Schluckstörung mit Speichelfluss
    \end{itemize}
  }
  \noteWithNote{Insektenstich}{\color{cDrug} $\rightarrow$SDH+Suprarenin}{%
    \begin{itemize}
      \item Schmerzen im Halsbereich, evtl. sichtbare Schwellung, Anamnese
    \end{itemize}
  }
  \drug{SDH}{Prednisolon}{%
    \dosierung{%

  }

    \end{kontraindikationen}
  }
