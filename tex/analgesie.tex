        {\color{cDrug}$\rightarrow$ \textbf{Perfalgan + Nalpain}}

  \drug{Buscopan}{Butylscopolamin}{%
    \dosierung{%
      \SI{0.3}{\milli\gram\per\kgKG} (\wdh{1} | \SI{5}{\minute}) (langsam!)
    }
    \begin{kontraindikationen}
      \item Glaukom
      \item Herzrythmusströrung
      \item Herzinsuffizienz
      \item Hyperthyreose
      \item (\mann) Prostatahyperplasie
      \item (\frau) Schwangerschaft / Stillen
      \item hereditäre \\ Fructoseintoleranz
      \item Megacolon
    \end{kontraindikationen}
  }

  \drug{Perfalgan}{Paracetamol}{%
    \dosierung{%
      \SI{1000}{\milli\gram} ({\erw} $\ge$\SI{50}{\kilo\gram})
      |
      \SI{15}{\milli\gram\per\kgKG} ({\kind} \SIrange{11}{50}{\kilo\gram})
    }
    \begin{kontraindikationen}
      \item G6PDH-Mangel
      \item Nieren-/Leberstörung
      \item chron./akuter C$_2$-Abusus
      \item Antiepileptika
      \item Paracetamol {\zeit} < \SI{4}{\hour}
      \item Tageshöchstdosis (\SI{70}{\milli\gram\per\kgKG})
    \end{kontraindikationen}
  }

  \drug{Nalpain}{Nalbuphin}{%
    \dosierung{%
      \SI{0.1}{\milli\gram\per\kgKG} (\wdh{1} | \SI{5}{\minute}) (langsam!)
    }
    \begin{kontraindikationen}
      \item Nieren-/Leberstörung
      \item chron./akuter C$_2$-Abusus
      \item Opiatabhängigkeit
      \item Einnahme {\zeit} < \SI{24}{\hour} \\
        Opiaten, Antihistaminika, Antidepressiva, Sedativa
    \end{kontraindikationen}
  }
